  <h2>和 ChatGPT 聊天</h2>

  <div id="chat"></div>

  <input id="input" type="text" placeholder="输入内容..." style="width: 80%;">

  <button onclick="sendMessage()">发送</button>


  <script>

    async function sendMessage() {

      const input = document.getElementById("input");

      const chat = document.getElementById("chat");

      const text = input.value;

      if (!text) return;


      chat.innerHTML += `<div class="msg user">你: ${text}</div>`;

      input.value = "";


      const response = await fetch("https://chatgpt-web-demo-bay.vercel.app/api/chat", {

        method: "POST",

        headers: { "Content-Type": "application/json" },

        body: JSON.stringify({ message: text })

      });


      const data = await response.json();

      chat.innerHTML += `<div class="msg bot">GPT: ${data.reply}</div>`;

      chat.scrollTop = chat.scrollHeight;

    }

  </script>
